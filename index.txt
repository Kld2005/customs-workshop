    <script>
        const ADMIN_PIN = "1234"; 
        
        let services = JSON.parse(localStorage.getItem('customs_services')) || [];
        let staff = JSON.parse(localStorage.getItem('customs_staff')) || [];

        const rankOrder = ["Owner", "Co Owner", "General Manager", "Supervisor", "Mechanic", "Trainee"];
        const rankLabels = {
            "Owner": "Owner - مالك الورشة", "Co Owner": "Co Owner - نائب المالك", "General Manager": "General Manager - المدير العام",
            "Supervisor": "Supervisor - مشرف", "Mechanic": "Mechanic - ميكانيكي", "Trainee": "Trainee - متدرب"
        };
        const rankClasses = {
            "Owner": "badge-owner", "Co Owner": "badge-coowner", "General Manager": "badge-manager",
            "Supervisor": "badge-supervisor", "Mechanic": "badge-mechanic", "Trainee": "badge-trainee"
        };

        function saveToLocalStorage() {
            localStorage.setItem('customs_services', JSON.stringify(services));
            localStorage.setItem('customs_staff', JSON.stringify(staff));
        }

        function switchPage(pageId) {
            document.querySelectorAll('.page').forEach(page => page.classList.remove('active'));
            document.getElementById(pageId).classList.add('active');
            
            if(pageId !== 'admin-auth-page') {
                document.getElementById('login-box').style.display = 'block';
                document.getElementById('admin-dashboard').style.display = 'none';
                document.getElementById('login-error').style.display = 'none';
                document.getElementById('admin-password').value = '';
                resetProductForm();
                resetStaffForm();
            }
        }

        function renderAll() {
            renderProducts();
            renderStaff();
        }

        // --- المنتجات ---
        function renderProducts() {
            const perfBody = document.getElementById('perf-table-body');
            const cosmBody = document.getElementById('cosm-table-body');
            const adminPBody = document.getElementById('admin-products-table');
            
            perfBody.innerHTML = ''; cosmBody.innerHTML = ''; adminPBody.innerHTML = '';
            let totalCivilianSum = 0; let totalMilitarySum = 0;
            let hasPerf = false; let hasCosm = false;

            if (services.length === 0) {
                const emptyMsg = `<tr><td colspan="3" class="empty-msg">لا توجد قطع مضافة حالياً.</td></tr>`;
                perfBody.innerHTML = emptyMsg; cosmBody.innerHTML = emptyMsg;
                adminPBody.innerHTML = `<tr><td colspan="5" class="empty-msg">لا توجد منتجات مضافة بعد.</td></tr>`;
            } else {
                services.forEach(service => {
                    const civPrice = parseFloat(service.civilian) || 0;
                    const milPrice = parseFloat(service.military) || 0;
                    totalCivilianSum += civPrice; totalMilitarySum += milPrice;

                    let catText = service.category === "performance" ? "أداء" : service.category === "cosmetic" ? "تجميلي" : "كلاهما";
                    const rowHTML = `<tr><td>${service.name}</td><td class="price-military txt-left">${milPrice.toLocaleString()}</td><td class="price-civilian txt-left">${civPrice.toLocaleString()}</td></tr>`;
                    
                    if (service.category === 'performance' || service.category === 'both') { perfBody.innerHTML += rowHTML; hasPerf = true; }
                    if (service.category === 'cosmetic' || service.category === 'both') { cosmBody.innerHTML += rowHTML; hasCosm = true; }

                    adminPBody.innerHTML += `
                        <tr>
                            <td>${service.name}</td>
                            <td style="color: #aaa; font-size:13px">${catText}</td>
                            <td class="price-military txt-left">${milPrice.toLocaleString()}</td>
                            <td class="price-civilian txt-left">${civPrice.toLocaleString()}</td>
                            <td class="txt-center">
                                <button class="btn-edit" onclick="startProductEdit(${service.id})">تعديل</button>
                                <button class="btn-delete" onclick="deleteProduct(${service.id})">حذف</button>
                            </td>
                        </tr>
                    `;
                });
                if(!hasPerf) perfBody.innerHTML = `<tr><td colspan="3" class="empty-msg">لا توجد منتجات أداء.</td></tr>`;
                if(!hasCosm) cosmBody.innerHTML = `<tr><td colspan="3" class="empty-msg">لا توجد منتجات تجميلية.</td></tr>`;
            }
            document.getElementById('total-civilian').innerText = totalCivilianSum.toLocaleString();
            document.getElementById('total-military').innerText = totalMilitarySum.toLocaleString();
        }

        function handleProductSubmit() {
            const id = document.getElementById('edit-product-id').value;
            const name = document.getElementById('new-pname').value;
            const civilian = document.getElementById('new-pcivilian').value;
            const military = document.getElementById('new-pmilitary').value;
            const category = document.getElementById('new-pcategory').value;

            if(!name || !civilian || !military) { alert('الرجاء تعبئة بيانات المنتج كاملاً!'); return; }

            if(id === "") {
                services.push({ id: Date.now(), name, civilian, military, category });
            } else {
                const idx = services.findIndex(s => s.id == id);
                if(idx !== -1) services[idx] = { id: Number(id), name, civilian, military, category };
                resetProductForm();
            }
            saveToLocalStorage(); 
            clearProductFields(); renderProducts();
        }

        function startProductEdit(id) {
            const s = services.find(p => p.id == id);
            if(s) {
                document.getElementById('edit-product-id').value = s.id;
                document.getElementById('new-pname').value = s.name;
                document.getElementById('new-pcivilian').value = s.civilian;
                document.getElementById('new-pmilitary').value = s.military;
                document.getElementById('new-pcategory').value = s.category;
                document.getElementById('product-form-title').innerText = "تعديل وتحديث المنتج الحالي";
                window.scrollTo({top: 0, behavior: 'smooth'});
            }
        }

        function resetProductForm() { document.getElementById('edit-product-id').value = ""; document.getElementById('product-form-title').innerText = "إضافة خدمة أو قطعة جديدة"; }
        function clearProductFields() { document.getElementById('new-pname').value = ''; document.getElementById('new-pcivilian').value = ''; document.getElementById('new-pmilitary').value = ''; }
        function deleteProduct(id) { services = services.filter(s => s.id !== id); saveToLocalStorage(); renderProducts(); }

        // --- الموظفين ---
        function renderStaff() {
            const clientStaffContainer = document.getElementById('staff-categories-container');
            const adminStaffBody = document.getElementById('admin-staff-table');
            clientStaffContainer.innerHTML = ''; adminStaffBody.innerHTML = '';

            rankOrder.forEach(rank => {
                const filteredStaff = staff.filter(member => member.rank === rank);
                if(filteredStaff.length > 0) {
                    let staffRows = '';
                    filteredStaff.forEach(member => {
                        staffRows += `<tr><td>${member.name}</td><td class="txt-center" style="color: #888;">${member.gameId}</td><td class="txt-left" style="color: #888; font-family: monospace;">${member.phone}</td></tr>`;
                    });

                    clientStaffContainer.innerHTML += `
                        <div class="staff-category-title ${rankClasses[rank]}">${rankLabels[rank]}</div>
                        <div class="table-container">
                            <table>
                                <thead><tr><th>اسم الموظف</th><th class="txt-center">رقم معرف الموظف (ID)</th><th class="txt-left">رقم الجوال</th></tr></thead>
                                <tbody>${staffRows}</tbody>
                            </table>
                        </div>`;
                }
            });

            if(staff.length === 0) {
                clientStaffContainer.innerHTML = `<div class="empty-msg" style="padding:40px;">لا يوجد موظفين مسجلين في الورشة حالياً.</div>`;
                adminStaffBody.innerHTML = `<tr><td colspan="5" class="empty-msg">لا يوجد موظفين مضافين.</td></tr>`;
            } else {
                staff.forEach(member => {
                    adminStaffBody.innerHTML += `
                        <tr>
                            <td><strong>${member.name}</strong></td>
                            <td class="${rankClasses[member.rank]}">${member.rank}</td>
                            <td>${member.gameId}</td>
                            <td style="font-family: monospace;">${member.phone}</td>
                            <td class="txt-center">
                                <button class="btn-edit" onclick="startStaffEdit(${member.id})">تعديل</button>
                                <button class="btn-delete" onclick="deleteStaff(${member.id})">حذف</button>
                            </td>
                        </tr>`;
                });
            }
        }

        function handleStaffSubmit() {
            const id = document.getElementById('edit-staff-id').value;
            const name = document.getElementById('staff-name').value;
            const gameId = document.getElementById('staff-game-id').value;
            const phone = document.getElementById('staff-phone').value;
            const rank = document.getElementById('staff-rank').value;

            if(!name || !gameId || !phone) { alert('الرجاء تعبئة كامل الحقول للموظف!'); return; }

            if(id === "") {
                staff.push({ id: Date.now(), name, gameId, phone, rank });
            } else {
                const idx = staff.findIndex(s => s.id == id);
                if(idx !== -1) staff[idx] = { id: Number(id), name, gameId, phone, rank };
                resetStaffForm();
            }
            saveToLocalStorage(); 
            clearStaffFields(); renderStaff();
        }

        function startStaffEdit(id) {
            const member = staff.find(s => s.id == id);
            if(member) {
                document.getElementById('edit-staff-id').value = member.id;
                document.getElementById('staff-name').value = member.name;
                document.getElementById('staff-game-id').value = member.gameId;
                document.getElementById('staff-phone').value = member.phone;
                document.getElementById('staff-rank').value = member.rank;
                document.getElementById('staff-form-title').innerText = "تعديل بيانات ورتبة الموظف الحالي";
                window.scrollTo({top: 150, behavior: 'smooth'});
            }
        }

        function resetStaffForm() { document.getElementById('edit-staff-id').value = ""; document.getElementById('staff-form-title').innerText = "إضافة / تعديل موظف بالورشة"; }
        function clearStaffFields() { document.getElementById('staff-name').value = ''; document.getElementById('staff-game-id').value = ''; document.getElementById('staff-phone').value = ''; }
        function deleteStaff(id) { staff = staff.filter(s => s.id !== id); saveToLocalStorage(); renderStaff(); }

        function checkPassword() {
            const enteredPass = document.getElementById('admin-password').value;
            if (enteredPass === ADMIN_PIN) {
                document.getElementById('login-box').style.display = 'none';
                document.getElementById('admin-dashboard').style.display = 'block';
                document.getElementById('login-error').style.display = 'none';
                document.getElementById('admin-password').value = '';
                renderAll();
            } else {
                document.getElementById('login-error').style.display = 'block';
            }
        }

        function logoutAdmin() { switchPage('welcome-page'); }

        renderAll();
